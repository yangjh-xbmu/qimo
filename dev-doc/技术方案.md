#### 2.2.1 话题验证逻辑 (utils/validators.ts)

```typescript
// 允许的话题关键词
const ALLOWED_TOPICS = [
  // 个人相关
  '林悦然', '悦然', '个人介绍', '背景', '经历', '教育',
  // 专业相关
  '戏剧', '电影', '影视', '导演', '编剧', '创作', '作品',
  '实验短片', '沉浸式戏剧', '跨媒介', '叙事', '表演',
  '《回声之间》', '《镜面剧场》', '《城市折叠》',
  // 学术相关
  '论文', '研究', '学术', '理论', '巴尔巴', '曼诺维奇',
  '铃木忠志', '福柯', '学者', '影响'
];

// 禁止的话题
const FORBIDDEN_TOPICS = [
  '政治', '宗教', '敏感', '暴力', '色情', '赌博',
  '投资', '理财', '医疗', '法律咨询', '其他人物'
];

export const validateTopic = (question: string): boolean => {
  const lowerQuestion = question.toLowerCase();
  
  // 检查是否包含禁止话题
  if (FORBIDDEN_TOPICS.some(topic => lowerQuestion.includes(topic.toLowerCase()))) {
    return false;
  }
  
  // 检查是否包含允许话题
  return ALLOWED_TOPICS.some(topic => 
    lowerQuestion.includes(topic.toLowerCase())
  );
};

export const getTopicHint = (): string => {
  return `您可以询问关于：
1. 我的个人背景与经历
2. 我的作品（如《回声之间》《镜面剧场》等）
3. 我的创作理念与研究方向
4. 我欣赏的学者及其影响
5. 戏剧与影视专业相关问题`;
};
```
